package com.cinema.dev.mappers;

import com.cinema.dev.dtos.UtilisateurRoleDTO;
import com.cinema.dev.models.Utilisateur;
import com.cinema.dev.models.Role;
import com.cinema.dev.models.UtilisateurRole;

public class UtilisateurRoleMap {

    public static UtilisateurRoleDTO toDTO(UtilisateurRole ur) {
        return new UtilisateurRoleDTO(
            ur.getId(),
            ur.getUtilisateur().getId(),
            ur.getRole().getId(),
            ur.getCreated()
        );
    }

    public static UtilisateurRole toEntity(UtilisateurRoleDTO dto, Utilisateur utilisateur, Role role) {
        UtilisateurRole ur = new UtilisateurRole();
        ur.setId(dto.getId());
        ur.setUtilisateur(utilisateur);
        ur.setRole(role);
        ur.setCreated(dto.getCreated());
        return ur;
    }
}
