package com.cinema.dev.mappers;

import com.cinema.dev.dtos.RoleDTO;
import com.cinema.dev.models.Role;

public class RoleMapper {

    public static RoleDTO toDTO(Role role) {
        return new RoleDTO(role.getId(), role.getLibelle());
    }

    public static Role toEntity(RoleDTO dto) {
        Role role = new Role();
        role.setId(dto.getId());
        role.setLibelle(dto.getLibelle());
        return role;
    }
}
